<!-- Key CPython ast nodes needed in this plan. Reference: stdlib ast module, Python 3.11+ -->

Statement nodes (ast.stmt subclasses):
  ast.Module(body: list[stmt], type_ignores: [])
  ast.Assign(targets: list[expr], value: expr)   ← no type annotation
  ast.Expr(value: expr)                           ← wraps a call expression for print/input at stmt position
  ast.If(test: expr, body: list[stmt], orelse: list[stmt])
  ast.While(test: expr, body: list[stmt], orelse: [])
  ast.For(target: expr, iter: expr, body: list[stmt], orelse: [])

Expression nodes (ast.expr subclasses):
  ast.Name(id: str, ctx: ast.Load()|ast.Store()|ast.Del())
  ast.Constant(value: int|str|bool|None)
  ast.Call(func: expr, args: list[expr], keywords: [])
  ast.BinOp(left: expr, op: ast.operator, right: expr)
    operators: ast.Add(), ast.Sub(), ast.Mult(), ast.Div()
  ast.Compare(left: expr, ops: list[cmpop], comparators: list[expr])
    cmpops: ast.Eq(), ast.NotEq(), ast.Lt(), ast.LtE(), ast.Gt(), ast.GtE()
  ast.BoolOp(op: ast.And()|ast.Or(), values: list[expr])   ← N-ary: (a and b and c)
  ast.UnaryOp(op: ast.Not()|ast.USub(), operand: expr)

Helpers:
  ast.fix_missing_locations(node) → fills lineno/col_offset with 1/0 defaults
  ast.unparse(node) → str

Atena node → CPython ast mapping (core constructs):
  Show(value) → ast.Expr(ast.Call(func=ast.Name("print"), args=[emit(value)]))
  Ask(prompt, target) → ast.Assign(targets=[ast.Name(_mangle(target), Store)],
                                   value=ast.Call(ast.Name("input"), [ast.Constant(prompt)]))
  Assign(name, value) → ast.Assign(targets=[ast.Name(_mangle(name), Store)], value=emit(value))
  If(cond, then, else) → ast.If(emit(cond), [emit(s) for s in then], [emit(s) for s in else])
  While(cond, body) → ast.While(emit(cond), [emit(s) for s in body], orelse=[])
  Repeat(count, body) → ast.For(
      target=ast.Name(f"_atena_i{self._loop_counter}", Store),
      iter=ast.Call(ast.Name("range"), [emit(count)]),
      body=[emit(s) for s in body], orelse=[])
      — increment _loop_counter BEFORE descending body; do NOT decrement after body.
      SUPERSEDED NOTE: The _emit_Repeat pattern in 04-PATTERNS.md shows a decrement
      (`self._loop_counter -= 1`) after the body. That pattern is SUPERSEDED here.
      The correct behavior is MONOTONIC: counter only ever increases, ensuring every
      repeat in the entire program gets a unique loop variable name regardless of nesting.
      Do NOT copy the decrement line from PATTERNS.md.
  BinOp(op, left, right):
    op "+" → ast.BinOp(emit(l), ast.Add(), emit(r))
    op "-" → ast.BinOp(emit(l), ast.Sub(), emit(r))
    op "*" → ast.BinOp(emit(l), ast.Mult(), emit(r))
    op "/" → ast.BinOp(emit(l), ast.Div(), emit(r))
    op "==" → ast.Compare(emit(l), [ast.Eq()], [emit(r)])
    op "!=" → ast.Compare(emit(l), [ast.NotEq()], [emit(r)])
    op "<"  → ast.Compare(emit(l), [ast.Lt()], [emit(r)])
    op "<=" → ast.Compare(emit(l), [ast.LtE()], [emit(r)])
    op ">"  → ast.Compare(emit(l), [ast.Gt()], [emit(r)])
    op ">=" → ast.Compare(emit(l), [ast.GtE()], [emit(r)])
    op "and" → ast.BoolOp(ast.And(), [emit(l), emit(r)])
    op "or"  → ast.BoolOp(ast.Or(), [emit(l), emit(r)])
  UnaryOp(op, operand):
    op "not" → ast.UnaryOp(ast.Not(), emit(operand))
    op "-"   → ast.UnaryOp(ast.USub(), emit(operand))
  Identifier(name) → ast.Name(_mangle(name), ast.Load())
  NumberLiteral(value) → ast.Constant(value)
  StringLiteral(value) → ast.Constant(value)   ← double-quote patch restores quotes post-unparse
  BoolLiteral(value) → ast.Constant(value)      ← True/False as Python bool constants

FunctionCall (plain user-defined + built-ins; no helper bodies yet):
  FunctionCall(name, args) → ast.Call(ast.Name(_mangle(name), Load), [emit(a) for a in args])
  Special case: name == "str" → ast.Call(ast.Name("str", Load), [...])
  Special case: name == "_atena_concat" → record in self._used_helpers; emit as plain call
  Special case: name == "_atena_index" → record in self._used_helpers; emit as plain call
