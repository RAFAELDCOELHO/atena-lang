       - _current(self) -> Token: returns self._tokens[self._pos] or self._tokens[-1] (EOF sentinel)
       - _peek(self) -> Token: returns self._tokens[self._pos + 1] or self._tokens[-1]
       - _advance(self) -> Token: consume and return current; do not advance past EOF
       - _check(self, *types: TokenType) -> bool: current.type in types
       - _match(self, *types: TokenType) -> Token | None: consume if matches, else None
       - _expect(self, tok_type: TokenType, message: str) -> Token: advance if matches, else raise _ParseError(current.line, message, current.source_line)

    8. _synchronize(self) -> None: per 02-PATTERNS.md §error-recovery-synchronize. Advance until NEWLINE or DEDENT; consume the sync token; progress invariant — always consumes ≥1 token.
