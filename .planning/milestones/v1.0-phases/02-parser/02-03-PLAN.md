    _dispatch_statement(self) -> Node:
    Peek the current token. Dispatch:
    - KEYWORD "show" → _parse_show()
    - KEYWORD "if" → _parse_if()
    - KEYWORD "while" → _parse_while()
    - KEYWORD "repeat" → _parse_repeat()
    - KEYWORD "function" → _parse_function_def()
    - KEYWORD "return" → _parse_return()
    - KEYWORD "add" → _parse_list_add()
    - KEYWORD "remove" → _parse_list_remove()
    - KEYWORD "ask" (bare, NOT as assignment RHS) → raise _ParseError for D-02 misuse (message from CONTEXT.md specifics: 'ask needs to save its answer into a name — try: answer = ask "What is your name?".')
    - NEWLINE → _advance() and return None (stray blank-line token; outer parse() filters None)
    - IDENTIFIER → may be assignment (name = ...) or bare function call (name(...)). Check _peek():
      if _peek() is ASSIGN → _parse_assignment()
      else → parse as expression statement (_parse_expression_statement())
    - Anything else → raise _ParseError(current.line, f'I didn\'t expect "{current.value}" here.', current.source_line)

    _parse_show(self) -> Show:
    kw = _advance()  # consume "show"
    value = _parse_expression()
    _consume_newline()
    return Show(value=value, line=kw.line, source_line=kw.source_line)

    _parse_assignment(self) -> Assign | Ask:
    name_tok = _advance()  # consume IDENTIFIER (the name)
    eq_tok = _advance()    # consume ASSIGN "="
    # Check if RHS starts with "ask" keyword:
    if _check(KEYWORD) and _current().value == "ask":
        return _parse_ask(target=name_tok.value, line=name_tok.line, source_line=name_tok.source_line)
    # Check for dot-write: e.g. "student.grade = 10" — handled differently:
    # Actually at this point we already consumed the IDENTIFIER. If the next token after ASSIGN
    # is a value expression, it's a plain Assign. For dot-write (student.grade = 10), the PARSER
    # handles it at statement level: peek IDENTIFIER DOT IDENTIFIER ASSIGN pattern.
    # For v1.0, handle "name = expr" simply: parse expression, emit Assign.
    value = _parse_expression()
    _consume_newline()
    return Assign(name=name_tok.value, value=value, line=name_tok.line, source_line=name_tok.source_line)

    _parse_ask(self, target: str, line: int, source_line: str) -> Ask:
    kw = _advance()  # consume "ask"
    if not _check(STRING):
        raise _ParseError(kw.line, 'ask needs to save its answer into a name — try: answer = ask "What is your name?".', kw.source_line)
    prompt_tok = _advance()  # consume STRING token
    _consume_newline()
    # prompt_tok.value is the literal string content (without outer double-quotes, if lexer strips them)
    return Ask(prompt=prompt_tok.value, target=target, line=line, source_line=source_line)

    _parse_if(self) -> If:
    kw = _advance()  # consume "if"
    condition = _parse_expression()
    _consume_newline()
    then_body = _parse_block()
    else_body: list[Node] = []
    # Optional else clause:
    if _check(KEYWORD) and _current().value == "else":
        _advance()  # consume "else"
        _consume_newline()
        else_body = _parse_block()
    return If(condition=condition, then_body=then_body, else_body=else_body, line=kw.line, source_line=kw.source_line)

    _parse_while(self) -> While:
    kw = _advance()  # consume "while"
    condition = _parse_expression()
    _consume_newline()
    body = _parse_block()
    return While(condition=condition, body=body, line=kw.line, source_line=kw.source_line)

    _parse_repeat(self) -> Repeat:
    kw = _advance()  # consume "repeat"
    count = _parse_expression()
    # Expect "times" keyword:
    if not (_check(KEYWORD) and _current().value == "times"):
        raise _ParseError(_current().line, '"repeat" needs the word "times" — try: repeat 5 times.', _current().source_line)
    _advance()  # consume "times"
    _consume_newline()
    body = _parse_block()
    return Repeat(count=count, body=body, line=kw.line, source_line=kw.source_line)

    _parse_function_def(self) -> FunctionDef:
    kw = _advance()  # consume "function"
    name_tok = _expect(IDENTIFIER, 'Expected a function name after "function".')
    _expect(LPAREN, f'Expected "(" after the function name "{name_tok.value}".')
    params: list[str] = []
    if not _check(RPAREN):
        params.append(_expect(IDENTIFIER, 'Expected a parameter name.').value)
        while _match(COMMA):
            params.append(_expect(IDENTIFIER, 'Expected a parameter name after ",".').value)
    _expect(RPAREN, 'I reached the end of the line still waiting for a ")".')
    _consume_newline()
    self._fn_depth += 1
    try:
        body = _parse_block()
    finally:
        self._fn_depth -= 1
    return FunctionDef(name=name_tok.value, params=params, body=body, line=kw.line, source_line=kw.source_line)

    _parse_return(self) -> Return:
    kw = _advance()  # consume "return"
    if self._fn_depth == 0:
        raise _ParseError(kw.line, '"return" only works inside a function.', kw.source_line)
    value = _parse_expression()
    _consume_newline()
    return Return(value=value, line=kw.line, source_line=kw.source_line)

    _parse_list_add(self) -> ListAdd:
    kw = _advance()  # consume "add"
    value = _parse_expression()
    _expect(KEYWORD, 'Expected "to" after the value in "add … to …".', value_check="to")
    target_tok = _expect(IDENTIFIER, 'Expected a list name after "to".')
    _consume_newline()
    return ListAdd(target=target_tok.value, value=value, line=kw.line, source_line=kw.source_line)

    _parse_list_remove(self) -> ListRemove:
    kw = _advance()  # consume "remove"
    value = _parse_expression()
    _expect(KEYWORD, 'Expected "from" after the value in "remove … from …".', value_check="from")
    target_tok = _expect(IDENTIFIER, 'Expected a list name after "from".')
    _consume_newline()
    return ListRemove(target=target_tok.value, value=value, line=kw.line, source_line=kw.source_line)

    Note on _expect with value_check: the current _expect() only checks token type. For "to"/"from"
    keywords (which lex as KEYWORD), use two steps: tok = _expect(KEYWORD, message); if tok.value != "to":
    raise _ParseError(tok.line, message, tok.source_line). Or add an optional value_check parameter
    to _expect() — executor's discretion, but keep it simple.

    _parse_expression_statement(self) -> FunctionCall | None:
    Parse a bare expression at statement position. Only function calls are valid here (per Claude's
    Discretion in CONTEXT.md: no meaningless bare-expression statements). Call _parse_expression();
    if result is not a FunctionCall, raise _ParseError(result.line, f'I didn\'t expect "{..." here.',
    result.source_line). _consume_newline(). Return the FunctionCall.

    _consume_newline(self) -> None:
    Consume a NEWLINE token if present; do nothing if at DEDENT/EOF (block-final statements may not
    have a trailing NEWLINE before DEDENT). This avoids failing on syntactically valid programs where
    the trailing NEWLINE was consumed by the block parser or not present.

    Important: _parse_assignment must handle the IDENTIFIER already consumed before we know it's an
    assignment. The dispatcher peeks _peek() for ASSIGN before dispatching to _parse_assignment, so
    by the time _parse_assignment runs, the name IDENTIFIER is still unconsumed (dispatcher peeked,
    did not consume). Revise the dispatcher flow accordingly:
    - IDENTIFIER peek: if _peek().type == ASSIGN → call _parse_assignment()
      where _parse_assignment starts by consuming IDENTIFIER then ASSIGN.

    Dot-write case (e.g., "student.grade = 10"):
    This is "IDENTIFIER DOT IDENTIFIER ASSIGN expr" — the lexer makes this: IDENTIFIER "student",
    DOT, IDENTIFIER "grade", ASSIGN, expr. The dispatcher sees IDENTIFIER first. _peek() sees DOT,
    not ASSIGN. So it falls through to _parse_expression_statement. But dot-write is a valid Atena
    statement. Handle it as: if _parse_expression() returns a DotAccess and next token is ASSIGN:
    consume ASSIGN; parse RHS; emit Assign(name=dotaccess_serialized, value=rhs) — or better, keep
    dot-write as a special case in the expression-statement path: after _parse_expression(), if result
    is DotAccess and current is ASSIGN: parse RHS, return Assign(name=f"{target}.{name}", value=rhs, ...).
    The generator handles the serialized "target.field" name → subscript write. Document this choice
    in the code comment. Alternative: use a separate DotWrite AST node — but ast_nodes.py has none, so
    reuse Assign with a compound name or produce DotAccess with an optional assigned_value. Simplest:
    parse "student.grade = 10" as Assign(name="student.grade", value=NumberLiteral(10)) — generator
    in Phase 4 splits the name on "." to emit student["grade"] = 10. Note this choice in a comment.
